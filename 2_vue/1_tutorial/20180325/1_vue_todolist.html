Vue.component('example', { props: { // 기본 타입 확인 ('null'은 어떤 타입이든 가능하다는 의미) propA: Number, // 여러 개의 가능한 타입 propB: [String, Number], // 문자열 + 필수값 propC: { type: String, required: true }, // 숫자 + default propD: { type: Number, default: 100 }, // 객체/배열의 기본값은 함수 사용해야 함 propE: { type: Object, default: function() { return {message : 'hello'} } }, // 사용자 정의 유효성 검사 기능 propF: { validator: function(value) { return value > 10 } } } })
